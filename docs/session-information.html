pkgs <- c("base", "Biobase", "BiocGenerics", "BiocStyle", "datasets",

          "graphics", "grDevices", "gridExtra", "IRanges", "lubridate",
          "magrittr", "MatrixGenerics", "matrixStats", "methods", "parallel",

   install.packages("BiocManager")
